      INT 21H

      INT 21H

      INT 21h

      MOV AH,9
      INT 21H
      
      MOV BL,CL

      INT 21h

      INT 21H

      add BL,30h

      add AL,30h
      MOV DL,AL;STORE ADDITION VALUE

      INT 21h 
      
      
;;;;;Division

        MOV AH,0h
        MOV AL,BL
         
       ;;;
      DIV Bh
