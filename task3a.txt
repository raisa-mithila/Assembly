.MODEL SMALL
.STACK 100H
.DATA
   a db "Enter a character: $"
.CODE
MAIN PROC
;iniitialize DS
MOV AX,@DATA
MOV DS,AX
;enter your code here   
    
    lea DX,a
    mov AH,9
    int 21h
             
    ; Read a character
    MOV AH, 1
    INT 21h 
    MOV bl,al
    

    ; Display the read character
    MOV AH, 2
    MOV DL,bl
    INT 21h
           
             
;exit to DOS
MOV AX,4C00H
INT 21H
MAIN ENDP
END MAIN
